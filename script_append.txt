<script>
        const escapeHtml = (unsafe) => {
            if(!unsafe) return '';
            return unsafe.toString()
                 .replace(/&/g, "&amp;")
                 .replace(/</g, "&lt;")
                 .replace(/>/g, "&gt;")
                 .replace(/"/g, "&quot;")
                 .replace(/'/g, "&#039;");
        };

        let currentHandoffInterval = null;
        let globalFeedbackData = [];
        let currentFeedbackFilter = 'ALL';

        function showErrorGlobal(msg) {
            const el = document.createElement('div');
            el.style.position = 'fixed';
            el.style.bottom = '20px';
            el.style.right = '20px';
            el.style.background = '#DC2626';
            el.style.color = 'white';
            el.style.padding = '10px 16px';
            el.style.borderRadius = '6px';
            el.style.fontSize = '12px';
            el.style.zIndex = '9999';
            el.innerText = msg;
            document.body.appendChild(el);
            setTimeout(() => { if(el.parentNode) el.parentNode.removeChild(el); }, 5000);
        }

        window.addEventListener('unhandledrejection', function(event) {
            console.error('DOLPHI ADMIN UNHANDLED ERROR:', event.reason);
            showErrorGlobal('An unexpected error occurred. Check the console for details.');
        });

        async function apiCall(endpoint, options = {}) {
            try {
                const res = await fetch(endpoint, options);
                if (res.status === 401) {
                    window.location.href = '/admin/login';
                    throw new Error('Unauthorized');
                }
                if (!res.ok) throw new Error('API Error: ' + res.status);
                return await res.json();
            } catch (err) {
                throw err;
            }
        }

        function showLoading(containerId) {
            const el = document.getElementById(containerId);
            if(el) {
                if (el.tagName === 'TBODY') {
                    const cols = el.parentElement.querySelectorAll('th').length || 1;
                    el.innerHTML = `<tr><td colspan="${cols}" style="text-align:center; color:#94A3B8; font-size:13px; padding:20px;">Loading...</td></tr>`;
                } else {
                    el.innerHTML = '<div style="text-align:center; color:#94A3B8; font-size:13px; padding:20px;">Loading...</div>';
                }
            }
        }

        function showError(containerId, message, retryFnArgs) {
            const el = document.getElementById(containerId);
            if(el) {
                if (el.tagName === 'TBODY') {
                    const cols = el.parentElement.querySelectorAll('th').length || 1;
                    el.innerHTML = `<tr><td colspan="${cols}" style="text-align:center; color:#DC2626; font-size:13px; padding:20px;">${message}<br><button class="action-btn" style="margin-top:10px" onclick="${retryFnArgs}">RETRY</button></td></tr>`;
                } else {
                    el.innerHTML = `<div style="text-align:center; color:#DC2626; font-size:13px; padding:20px;">${message}<br><button class="action-btn" style="margin-top:10px" onclick="${retryFnArgs}">RETRY</button></div>`;
                }
            }
        }

        async function loadStats() {
            try {
                const data = await apiCall('/api/admin/stats');
                document.getElementById('stat-msgs').innerText = data.total_messages_today;
                document.getElementById('stat-sessions').innerText = data.total_sessions;
                document.getElementById('stat-conf').innerText = data.average_confidence + '%';
                document.getElementById('stat-esc').innerText = data.escalation_rate + '%';
            } catch(e) {
                showErrorGlobal('Failed to load stats details');
            }
        }

        async function loadSettings() {
            try {
                const s = await apiCall('/api/admin/settings');
                document.getElementById('set-code').value = s.adminAccessCode || '';
                document.getElementById('set-base').value = s.baseUrl || '';
            } catch(e) {
                console.error(e);
            }
        }

        async function loadUsers() {
            showLoading('users-table-body');
            try {
                const data = await apiCall('/api/admin/users');
                const tbody = document.getElementById('users-table-body');
                tbody.innerHTML = data.map(u => {
                    const badge = u.role === 'admin' ? `<span style="background:#1A3A5C; color:white; padding:2px 6px; border-radius:10px; font-size:10px; font-weight:bold;">Admin</span>` : `<span style="background:#94A3B8; color:white; padding:2px 6px; border-radius:10px; font-size:10px; font-weight:bold;">User</span>`;
                    const statusText = u.is_active ? 'Active' : 'Inactive';
                    const statusColor = u.is_active ? '#16A34A' : '#DC2626';
                    return `
                    <tr>
                        <td>${u.id}</td>
                        <td>${escapeHtml(u.username)}</td>
                        <td>${escapeHtml(u.email)}</td>
                        <td>${badge} ${u.used_access_code ? '<span style="background:#F5C518; color:#1A3A5C; padding:2px 6px; border-radius:10px; font-size:10px; font-weight:bold; margin-left:6px;">Access Code</span>' : ''}</td>
                        <td>${new Date(u.created_at).toLocaleString()}</td>
                        <td>${u.last_login ? new Date(u.last_login).toLocaleString() : 'Never'}</td>
                        <td style="color:${statusColor}; font-weight:bold;">${statusText}</td>
                        <td>
                            <button class="action-btn" onclick="toggleUserDetails(${u.id})">VIEW</button>
                            <button class="action-btn amber" onclick="toggleEditUser(${u.id})">EDIT</button>
                            <button class="action-btn ${u.is_active ? '' : 'amber'}" onclick="toggleUserActive(${u.id})">${u.is_active ? 'DEACTIVATE' : 'ACTIVATE'}</button>
                            <div id="user-err-${u.id}" class="field-error" style="display:none; margin-top:5px;"></div>
                        </td>
                    </tr>
                    <tr id="details-row-${u.id}" style="display:none; background:#F8FAFC;">
                        <td colspan="8" style="padding:16px;">
                            <div style="font-size:12px; color:#475569;">
                                <strong>Created By:</strong> ${u.created_by ? `Admin #${u.created_by}` : 'Self-registered'}<br>
                                <strong>Total Messages:</strong> ${u.message_count || 0}
                            </div>
                        </td>
                    </tr>
                    <tr id="edit-row-${u.id}" style="display:none; background:#F8FAFC;">
                        <td colspan="8" style="padding:16px;">
                            <div style="display:flex; gap:12px; align-items:center;">
                                <select id="eu-role-${u.id}" class="search-input" style="width:120px; margin:0;"><option value="user" ${u.role==='user'?'selected':''}>User</option><option value="admin" ${u.role==='admin'?'selected':''}>Admin</option></select>
                                <label style="font-size:12px; display:flex; align-items:center; gap:4px;"><input type="checkbox" id="eu-active-${u.id}" ${u.is_active ? 'checked' : ''}> Active</label>
                                <input type="password" id="eu-pwd-${u.id}" class="search-input" placeholder="New Password (optional)" style="width:180px; margin:0;">
                                <button class="action-btn" onclick="saveUser(${u.id})">SAVE</button>
                                <button class="action-btn" style="background:#94A3B8;" onclick="toggleEditUser(${u.id})">CANCEL</button>
                            </div>
                        </td>
                    </tr>
                    `;
                }).join('');
            } catch(e) {
                showError('users-table-body', 'Failed to load data. Please try again.', 'loadUsers()');
            }
        }

        window.toggleUserDetails = function(id) {
            const el = document.getElementById('details-row-' + id);
            if(el) el.style.display = el.style.display === 'none' ? 'table-row' : 'none';
        }

        window.toggleEditUser = function(id) {
            const el = document.getElementById('edit-row-' + id);
            if(el) el.style.display = el.style.display === 'none' ? 'table-row' : 'none';
        }

        window.toggleUserActive = async function(id) {
            try {
                await apiCall('/api/admin/users/' + id + '/deactivate', { method: 'PATCH' });
                loadUsers();
            } catch(e) {
                const el = document.getElementById('user-err-' + id);
                if(el) { el.innerText = e.message; el.style.display = 'block'; }
            }
        }

        window.saveUser = async function(id) {
            try {
                const role = document.getElementById('eu-role-' + id).value;
                const is_active = document.getElementById('eu-active-' + id).checked;
                const pwd = document.getElementById('eu-pwd-' + id).value;
                const payload = { role, is_active };
                if (pwd) payload.password = pwd;
                await apiCall('/api/admin/users/' + id, {
                    method: 'PATCH',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify(payload)
                });
                loadUsers();
            } catch(e) {
                const el = document.getElementById('user-err-' + id);
                if(el) { el.innerText = e.message; el.style.display = 'block'; }
            }
        }

        window.toggleCreateUser = function() {
            const el = document.getElementById('create-user-form');
            if(el) el.style.display = el.style.display === 'none' ? 'block' : 'none';
        }

        window.createUser = async function() {
            try {
                const full_name = document.getElementById('cu-fullname').value;
                const username = document.getElementById('cu-username').value;
                const email = document.getElementById('cu-email').value;
                const password = document.getElementById('cu-password').value;
                const role = document.getElementById('cu-role').value;
                await apiCall('/api/admin/users/create', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ full_name, username, email, password, role })
                });
                toggleCreateUser();
                document.getElementById('cu-fullname').value = '';
                document.getElementById('cu-username').value = '';
                document.getElementById('cu-email').value = '';
                document.getElementById('cu-password').value = '';
                loadUsers();
            } catch(e) {
                alert('Failed to create user: ' + e.message);
            }
        }

        async function loadKb() {
            showLoading('kb-table-body');
            try {
                const data = await apiCall('/api/documents');
                const tbody = document.getElementById('kb-table-body');
                tbody.innerHTML = data.map(d => `
                    <tr>
                        <td>${escapeHtml(d.filename)}</td>
                        <td>${escapeHtml(d.file_type || d.source_type)}</td>
                        <td>${d.chunk_count || d.chunks_count || 0} chunks</td>
                        <td>${new Date(d.upload_timestamp).toLocaleString()}</td>
                        <td><button class="action-btn" style="background:#DC2626;" onclick="deleteDoc(${d.id})">DELETE</button></td>
                    </tr>
                `).join('');
            } catch(e) {
                showError('kb-table-body', 'Failed to load data. Please try again.', 'loadKb()');
            }
        }

        window.deleteDoc = async function(id) {
            if(!confirm('Are you sure?')) return;
            try {
                await apiCall('/api/documents/' + id, { method: 'DELETE' });
                loadKb();
            } catch(e) {
                alert('Delete failed: ' + e.message);
            }
        }

        window.uploadDocuments = async function() {
            const fileInput = document.getElementById('kb-file');
            const statusDiv = document.getElementById('upload-status');
            if (fileInput.files.length === 0) return;
            statusDiv.innerText = 'Ingesting... please wait';
            
            const formData = new FormData();
            for (let i = 0; i < fileInput.files.length; i++) {
                formData.append('document', fileInput.files[i]);
            }
            try {
                const res = await fetch('/api/upload-document', {
                    method: 'POST',
                    body: formData
                });
                if (!res.ok) throw new Error('Upload failed');
                statusDiv.innerText = 'Upload successful!';
                setTimeout(() => statusDiv.innerText = '', 3000);
                fileInput.value = '';
                loadKb();
            } catch(e) {
                statusDiv.innerText = 'Error: ' + e.message;
            }
        }

        let convPage = 1;
        let convAllData = [];
        async function loadConversations(page=1) {
            convPage = page;
            showLoading('conv-table-body');
            try {
                const data = await apiCall('/api/admin/conversations?page=' + page + '&per_page=20');
                convAllData = data;
                renderConversations(data);
            } catch(e) {
                showError('conv-table-body', 'Failed to load data. Please try again.', 'loadConversations(' + page + ')');
            }
        }

        function renderConversations(data) {
            const tbody = document.getElementById('conv-table-body');
            tbody.innerHTML = data.map(c => {
                const truncId = c.session_id.substring(0, 12);
                return `
                <tr>
                    <td>${escapeHtml(truncId)}</td>
                    <td>${new Date(c.timestamp).toLocaleString()}</td>
                    <td>${c.msg_count || 0}</td>
                    <td><button class="action-btn" onclick="viewChat('${escapeHtml(c.session_id)}')">VIEW</button></td>
                </tr>
                <tr id="chat-row-${escapeHtml(c.session_id)}" style="display:none;">
                    <td colspan="4"><div id="chat-${escapeHtml(c.session_id)}" class="chat-container" style="max-height: 300px; overflow-y: auto; padding: 12px; background: #F8FAFC; border-top: 1px solid #E2E8F0;">Loading...</div></td>
                </tr>
                `;
            }).join('');
        }

        window.filterConversations = function() {
            const val = document.getElementById('conv-search').value.toLowerCase();
            const filtered = convAllData.filter(c => c.session_id.toLowerCase().includes(val));
            renderConversations(filtered);
        }

        window.viewChat = async function(sessionId) {
            const row = document.getElementById('chat-row-' + sessionId);
            const container = document.getElementById('chat-' + sessionId);
            if(row.style.display === 'table-row') {
                row.style.display = 'none';
                return;
            }
            row.style.display = 'table-row';
            container.innerHTML = '<div style="font-size:12px; color:#94A3B8;">Loading transcript...</div>';
            try {
                const msgs = await apiCall('/api/admin/conversations/' + sessionId + '/messages');
                container.innerHTML = msgs.map(m => {
                    const roleLabel = m.role === 'user' ? 'USER' : 'DOLPHI';
                    let txt = m.content;
                    if(window.marked) txt = window.marked.parse(txt);
                    return `<div style="margin-bottom:10px; font-size:13px;"><strong>${roleLabel}:</strong> <span style="color:#94A3B8; font-size:10px;">(${new Date(m.timestamp).toLocaleTimeString()})</span><div style="margin-top:4px;">${txt}</div></div>`;
                }).join('');
            } catch(e) {
                container.innerHTML = '<div style="color:#DC2626; font-size:12px;">Failed to load messages</div>';
            }
        }

        async function loadHandoffs() {
            showLoading('hq-table-body');
            try {
                const data = await apiCall('/api/admin/handoffs');
                renderHandoffs(data);
            } catch(e) {
                showError('hq-table-body', 'Failed to load data. Please try again.', 'loadHandoffs()');
            }
        }

        function renderHandoffs(data) {
            const tbody = document.getElementById('hq-table-body');
            tbody.innerHTML = data.map(h => {
                const truncId = h.session_id.substring(0, 12);
                const badgeStyle = h.status === 'resolved' ? 'color:#16A34A;' : 'color:#F5C518; font-weight:bold;';
                return `
                <tr>
                    <td>${h.id}</td>
                    <td>${escapeHtml(truncId)}</td>
                    <td>${new Date(h.triggered_at).toLocaleString()}</td>
                    <td><span id="ho-status-${h.id}" style="${badgeStyle}">${escapeHtml(h.status)}</span></td>
                    <td>
                        <button class="action-btn" onclick="resolveHandoff(${h.id})">RESOLVE</button>
                        <button class="action-btn" onclick="viewHandoffChat('${escapeHtml(h.session_id)}')">VIEW CHAT</button>
                    </td>
                </tr>
                <tr id="ho-chat-row-${escapeHtml(h.session_id)}" style="display:none;">
                    <td colspan="5"><div id="ho-chat-${escapeHtml(h.session_id)}" class="chat-container" style="max-height: 300px; overflow-y: auto; padding: 12px; background: #F8FAFC; border-top: 1px solid #E2E8F0;">Loading...</div></td>
                </tr>
                `;
            }).join('');
        }

        window.resolveHandoff = async function(id) {
            try {
                await apiCall('/api/admin/handoff/' + id, {
                    method: 'PATCH',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ status: 'resolved' })
                });
                const st = document.getElementById('ho-status-' + id);
                if(st) { st.innerText = 'resolved'; st.style.color = '#16A34A'; st.style.fontWeight = 'normal'; }
            } catch(e) {
                alert('Resolve failed: ' + e.message);
            }
        }

        window.viewHandoffChat = async function(sessionId) {
            const row = document.getElementById('ho-chat-row-' + sessionId);
            const container = document.getElementById('ho-chat-' + sessionId);
            if(row.style.display === 'table-row') {
                row.style.display = 'none';
                return;
            }
            row.style.display = 'table-row';
            container.innerHTML = '<div style="font-size:12px; color:#94A3B8;">Loading transcript...</div>';
            try {
                const msgs = await apiCall('/api/admin/conversations/' + sessionId + '/messages');
                container.innerHTML = msgs.map(m => {
                    const roleLabel = m.role === 'user' ? 'USER' : 'DOLPHI';
                    let txt = m.content;
                    if(window.marked) txt = window.marked.parse(txt);
                    return `<div style="margin-bottom:10px; font-size:13px;"><strong>${roleLabel}:</strong> <div style="margin-top:4px;">${txt}</div></div>`;
                }).join('');
            } catch(e) {
                container.innerHTML = '<div style="color:#DC2626; font-size:12px;">Failed to load messages</div>';
            }
        }

        async function loadFeedback() {
            showLoading('fb-table-body');
            try {
                const data = await apiCall('/api/admin/feedback');
                globalFeedbackData = data;
                renderFeedback(data);
            } catch(e) {
                showError('fb-table-body', 'Failed to load data. Please try again.', 'loadFeedback()');
            }
        }

        function renderFeedback(data) {
            const tbody = document.getElementById('fb-table-body');
            const filtered = data.filter(d => {
                if(currentFeedbackFilter === 'ALL') return true;
                if(currentFeedbackFilter === 'POSITIVE') return d.rating > 0;
                if(currentFeedbackFilter === 'NEGATIVE') return d.rating < 0;
                return true;
            });
            tbody.innerHTML = filtered.map(d => {
                const truncId = d.session_id ? d.session_id.substring(0, 12) : 'N/A';
                const emoji = d.rating > 0 ? '👍' : '👎';
                let msgPreview = d.message_content || d.message || '';
                if(msgPreview.length > 80) msgPreview = msgPreview.substring(0, 80) + '...';
                return `
                <tr>
                    <td>${escapeHtml(truncId)}</td>
                    <td style="font-size:16px;">${emoji}</td>
                    <td style="max-width:300px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis;" title="${escapeHtml(msgPreview)}">${escapeHtml(msgPreview)}</td>
                    <td>${new Date(d.created_at).toLocaleString()}</td>
                </tr>
                `;
            }).join('');
        }

        window.setFeedbackFilter = function(filter, btnEl) {
            currentFeedbackFilter = filter;
            if(btnEl) {
                btnEl.parentElement.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
                btnEl.classList.add('active');
            }
            renderFeedback(globalFeedbackData);
        }

        document.addEventListener('DOMContentLoaded', () => {
            console.log('DOLPHI ADMIN: Initializing navigation');

            const navItems = document.querySelectorAll('.nav-item');
            navItems.forEach(item => {
                item.addEventListener('click', (e) => {
                    const tabId = item.getAttribute('data-tab');
                    if(!tabId) return;

                    navItems.forEach(el => el.classList.remove('active'));
                    item.classList.add('active');
                    
                    document.querySelectorAll('.section').forEach(el => {
                        el.style.display = 'none';
                        el.classList.remove('active');
                    });
                    
                    const sec = document.getElementById(tabId);
                    if(sec) {
                        sec.style.display = 'block';
                        sec.classList.add('active');
                    }

                    if(currentHandoffInterval) {
                        clearInterval(currentHandoffInterval);
                        currentHandoffInterval = null;
                    }

                    if (tabId === 'dashboard') loadStats();
                    else if (tabId === 'settings') loadSettings();
                    else if (tabId === 'knowledge_base') loadKb();
                    else if (tabId === 'conversations') loadConversations(1);
                    else if (tabId === 'handoff_queue') {
                        loadHandoffs();
                        currentHandoffInterval = setInterval(loadHandoffs, 30000);
                    }
                    else if (tabId === 'feedback') loadFeedback();
                    else if (tabId === 'users') loadUsers();
                });
            });

            document.querySelectorAll('.section').forEach(el => {
                if(!el.classList.contains('active')) el.style.display = 'none';
                else el.style.display = 'block';
            });
            loadStats();
        });
    </script>
</body>
</html>
